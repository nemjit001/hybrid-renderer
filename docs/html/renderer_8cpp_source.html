656void Renderer::prepareFrameResources(uint32_t frameIdx)
657{
658 assert(frameIdx < hri::RenderCore::framesInFlight());
659 m_renderCore.awaitFrameFinished(frameIdx); // Ensure this frame's execution has finished
660
661 const std::vector<RenderInstance>& renderInstanceList = m_activeScene.generateRenderInstanceList(m_camera);
662 RendererFrameData& rendererFrameData = m_frames[frameIdx];
663
664 // Update UBO staging buffers
665 hri::CameraShaderData cameraData = m_camera.getShaderData();
666 rendererFrameData.cameraUBO->copyToBuffer(&cameraData, sizeof(hri::CameraShaderData));
667
668 // Create or reallocate TLAS
669 if (rendererFrameData.tlas.get() == nullptr
670 || m_activeScene.accelStructManager.shouldReallocTLAS(*rendererFrameData.tlas, renderInstanceList, rendererFrameData.blasList)
671 )
672 {
673 rendererFrameData.tlas = std::make_unique<raytracing::AccelerationStructure>(
674 m_activeScene.accelStructManager.createTLAS(renderInstanceList, rendererFrameData.blasList)
675 );
676 }
677
678 // Build BLAS list & TLAS
679 VkCommandBuffer oneshot = m_computePool.createCommandBuffer();
680 m_activeScene.accelStructManager.cmdBuildBLASses(oneshot, renderInstanceList, m_activeScene.meshes, rendererFrameData.blasList);
681 m_activeScene.accelStructManager.cmdBuildTLAS(oneshot, renderInstanceList, rendererFrameData.blasList, *rendererFrameData.tlas);
682 m_computePool.submitAndWait(oneshot);
683 m_computePool.freeCommandBuffer(oneshot);
684
685 // Always write TLAS descriptor in case of updated TLAS structure
686 VkWriteDescriptorSetAccelerationStructureKHR tlasInfo = VkWriteDescriptorSetAccelerationStructureKHR{ VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET_ACCELERATION_STRUCTURE_KHR };
687 tlasInfo.accelerationStructureCount = 1;
688 tlasInfo.pAccelerationStructures = &rendererFrameData.tlas->accelerationStructure;
689
690 (*rendererFrameData.raytracingSet)
691 .writeEXT(RayTracingBindings::Tlas, &tlasInfo)
692 .flush();
693
694 // Update subsystem frame info
695 m_gbufferLayoutSubsystem->updateFrameInfo(GBufferLayoutFrameInfo{
696 rendererFrameData.sceneDataSet->set,
697 &m_activeScene,
698 });
699
700 m_hybridRTSubsystem->updateFrameInfo(RayTracingFrameInfo{
701 rendererFrameData.sceneDataSet->set,
702 rendererFrameData.raytracingSet->set,
703 });
704
705 m_presentSubsystem->updateFrameInfo(PresentFrameInfo{
706 rendererFrameData.presentInputSet->set,
