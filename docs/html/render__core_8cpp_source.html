104void RenderCore::endFrame()
105{
106 FrameState& activeFrame = m_frames[m_currentFrame];
107
108 // Can't do work if the swapchain needs to be recreated
109 if (m_recreateSwapchain)
110 return;
111
112 VkFence frameFences[] = { activeFrame.frameReady };
113 HRI_VK_CHECK(vkResetFences(m_ctx.device, HRI_SIZEOF_ARRAY(frameFences), frameFences));
114
115 VkPipelineStageFlags waitStages[] = {
116 VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT
117 };
118
119 VkSubmitInfo frameSubmit = VkSubmitInfo{ VK_STRUCTURE_TYPE_SUBMIT_INFO };
120 frameSubmit.commandBufferCount = 1;
121 frameSubmit.pCommandBuffers = &activeFrame.graphicsCommandBuffer;
122 frameSubmit.waitSemaphoreCount = 1;
123 frameSubmit.pWaitSemaphores = &activeFrame.imageAvailable;
124 frameSubmit.pWaitDstStageMask = waitStages;
125 frameSubmit.signalSemaphoreCount = 1;
126 frameSubmit.pSignalSemaphores = &activeFrame.renderingFinished;
127 HRI_VK_CHECK(vkQueueSubmit(m_ctx.queues.graphicsQueue.handle, 1, &frameSubmit, activeFrame.frameReady));
128
129 VkSwapchainKHR swapchains[] = { m_ctx.swapchain };
130 uint32_t imageIndices[] = { m_activeSwapImage };
131 assert(HRI_SIZEOF_ARRAY(swapchains) == HRI_SIZEOF_ARRAY(imageIndices));
132
133 VkPresentInfoKHR presentInfo = VkPresentInfoKHR{ VK_STRUCTURE_TYPE_PRESENT_INFO_KHR };
134 presentInfo.swapchainCount = HRI_SIZEOF_ARRAY(swapchains);
135 presentInfo.pSwapchains = swapchains;
136 presentInfo.pImageIndices = imageIndices;
137 presentInfo.pResults = nullptr;
138 presentInfo.waitSemaphoreCount = 1;
139 presentInfo.pWaitSemaphores = &activeFrame.renderingFinished;
140 validateSwapchainState(vkQueuePresentKHR(m_ctx.queues.presentQueue.handle, &presentInfo));
141
142 m_previousFrame = m_currentFrame;
143 m_currentFrame = (m_currentFrame + 1) % HRI_VK_FRAMES_IN_FLIGHT;
144}
