9FrameState FrameState::init(RenderContext* ctx)
10{
11 assert(ctx != nullptr);
12 FrameState frameState = FrameState{};
13
14 VkFenceCreateInfo frameReadyFenceCreateInfo = VkFenceCreateInfo{ VK_STRUCTURE_TYPE_FENCE_CREATE_INFO };

17
18 VkSemaphoreCreateInfo semaphoreCreateInfo = VkSemaphoreCreateInfo{ VK_STRUCTURE_TYPE_SEMAPHORE_CREATE_INFO };
19 semaphoreCreateInfo.flags = 0;
20 HRI_VK_CHECK(vkCreateSemaphore(ctx->device, &semaphoreCreateInfo, nullptr, &frameState.imageAvailable));
21 HRI_VK_CHECK(vkCreateSemaphore(ctx->device, &semaphoreCreateInfo, nullptr, &frameState.renderingFinished));
22
23 VkCommandPoolCreateInfo gfxPoolCreateInfo = VkCommandPoolCreateInfo{ VK_STRUCTURE_TYPE_COMMAND_POOL_CREATE_INFO };
24 gfxPoolCreateInfo.flags = VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT;
25 gfxPoolCreateInfo.queueFamilyIndex = ctx->queues.graphicsQueue.family;
26 HRI_VK_CHECK(vkCreateCommandPool(ctx->device, &gfxPoolCreateInfo, nullptr, &frameState.graphicsCommandPool));
27
28 VkCommandBufferAllocateInfo gfxBufferAllocateInfo = VkCommandBufferAllocateInfo{ VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO };
29 gfxBufferAllocateInfo.commandPool = frameState.graphicsCommandPool;
30 gfxBufferAllocateInfo.commandBufferCount = 1;
31 gfxBufferAllocateInfo.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
32 HRI_VK_CHECK(vkAllocateCommandBuffers(ctx->device, &gfxBufferAllocateInfo, &frameState.graphicsCommandBuffer));
33
34 return frameState;

37void FrameState::destroy(RenderContext* ctx, FrameState& frameState)
38{
39 assert(ctx != nullptr);
40 HRI_VK_CHECK(vkWaitForFences(ctx->device, 1, &frameState.frameReady, VK_TRUE, UINT64_MAX));
41
42 vkDestroyFence(ctx->device, frameState.frameReady, nullptr);
43 vkDestroySemaphore(ctx->device, frameState.imageAvailable, nullptr);
44 vkDestroySemaphore(ctx->device, frameState.renderingFinished, nullptr);
45 vkDestroyCommandPool(ctx->device, frameState.graphicsCommandPool, nullptr);
46
47 memset(&frameState, 0, sizeof(FrameState));
48}

168void RenderCore::startFrame()
169{
170 FrameState& activeFrame = m_frames[m_currentFrame];
171
172 VkFence frameFences[] = { activeFrame.frameReady };
173 HRI_VK_CHECK(vkWaitForFences(m_pCtx->device, HRI_SIZEOF_ARRAY(frameFences), frameFences, VK_TRUE, UINT64_MAX));
174
175 if (m_recreateSwapchain)
176 {
177 m_pCtx->recreateSwapchain();
178
179 if (m_onSwapchainInvalidateFunc != nullptr)
180 m_onSwapchainInvalidateFunc(m_pCtx->swapchain);
181
182 m_recreateSwapchain = false;
183 }
184
185 validateSwapchainState(vkAcquireNextImageKHR(m_pCtx->device, m_pCtx->swapchain, UINT64_MAX, activeFrame.imageAvailable, VK_NULL_HANDLE, &m_activeSwapImage));
186}

188void RenderCore::endFrame()
189{
190 FrameState& activeFrame = m_frames[m_currentFrame];
191
192 // Can't do work if the swapchain needs to be recreated
193 if (m_recreateSwapchain)
194 return;
195
196 VkFence frameFences[] = { activeFrame.frameReady };
197 HRI_VK_CHECK(vkResetFences(m_pCtx->device, HRI_SIZEOF_ARRAY(frameFences), frameFences));
198
199 VkPipelineStageFlags waitStages[] = {
200 VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT
201 };
202
203 VkSubmitInfo frameSubmit = VkSubmitInfo{ VK_STRUCTURE_TYPE_SUBMIT_INFO };
204 frameSubmit.commandBufferCount = 1;
205 frameSubmit.pCommandBuffers = &activeFrame.graphicsCommandBuffer;
206 frameSubmit.waitSemaphoreCount = 1;
207 frameSubmit.pWaitSemaphores = &activeFrame.imageAvailable;
208 frameSubmit.pWaitDstStageMask = waitStages;
209 frameSubmit.signalSemaphoreCount = 1;
210 frameSubmit.pSignalSemaphores = &activeFrame.renderingFinished;
211 HRI_VK_CHECK(vkQueueSubmit(m_pCtx->queues.graphicsQueue.handle, 1, &frameSubmit, activeFrame.frameReady));
212
213 VkSwapchainKHR swapchains[] = { m_pCtx->swapchain };
214 uint32_t imageIndices[] = { m_activeSwapImage };
215 assert(HRI_SIZEOF_ARRAY(swapchains) == HRI_SIZEOF_ARRAY(imageIndices));
216
217 VkPresentInfoKHR presentInfo = VkPresentInfoKHR{ VK_STRUCTURE_TYPE_PRESENT_INFO_KHR };
218 presentInfo.swapchainCount = HRI_SIZEOF_ARRAY(swapchains);
219 presentInfo.pSwapchains = swapchains;
220 presentInfo.pImageIndices = imageIndices;
221 presentInfo.pResults = nullptr;
222 presentInfo.waitSemaphoreCount = 1;
223 presentInfo.pWaitSemaphores = &activeFrame.renderingFinished;
224 validateSwapchainState(vkQueuePresentKHR(m_pCtx->queues.presentQueue.handle, &presentInfo));
225
226 m_previousFrame = m_currentFrame;
227 m_currentFrame = (m_currentFrame + 1) % HRI_VK_FRAMES_IN_FLIGHT;
228}

245void RenderCore::immediateSubmit(HRIImmediateSubmitFunc submitFunc)
246{
247 VkCommandBuffer oneshotBuffer = VK_NULL_HANDLE;
248
249 VkCommandBufferAllocateInfo oneshotAllocateInfo = VkCommandBufferAllocateInfo{ VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO };
250 oneshotAllocateInfo.commandPool = m_submitPool;
251 oneshotAllocateInfo.commandBufferCount = 1;
252 oneshotAllocateInfo.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
253 HRI_VK_CHECK(vkAllocateCommandBuffers(m_pCtx->device, &oneshotAllocateInfo, &oneshotBuffer));
254
255 VkCommandBufferBeginInfo beginOneshot = VkCommandBufferBeginInfo{ VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO };
256 beginOneshot.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
257 beginOneshot.pInheritanceInfo = nullptr;
258 vkBeginCommandBuffer(oneshotBuffer, &beginOneshot);
259 submitFunc(oneshotBuffer);
260 vkEndCommandBuffer(oneshotBuffer);
261
262 VkSubmitInfo oneshotSubmit = VkSubmitInfo{ VK_STRUCTURE_TYPE_SUBMIT_INFO };
263 oneshotSubmit.commandBufferCount = 1;
264 oneshotSubmit.pCommandBuffers = &oneshotBuffer;
265 // FIXME: Add fence based synchronization & use explicit queue for operation type!
266 HRI_VK_CHECK(vkQueueSubmit(m_pCtx->queues.graphicsQueue.handle, 1, &oneshotSubmit, VK_NULL_HANDLE));
267 vkDeviceWaitIdle(m_pCtx->device);
268
269 vkFreeCommandBuffers(m_pCtx->device, m_submitPool, 1, &oneshotBuffer);
270}
