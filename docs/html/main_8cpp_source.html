86void handleCameraInput(WindowHandle* window, float deltaTime, hri::Camera& camera)
87{
88 bool cameraUpdated = false;
89
90 hri::Float3 forward = camera.forward;
91 hri::Float3 right = camera.right;
92 hri::Float3 up = camera.up;
93
94 hri::Float3 positionDelta = hri::Float3(0.0f);
95
96 if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS) positionDelta += forward * 2.0f * CAMERA_SPEED * deltaTime; cameraUpdated = true;
97 if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS) positionDelta -= forward * 2.0f * CAMERA_SPEED * deltaTime; cameraUpdated = true;
98 if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS) positionDelta += right * 2.0f * CAMERA_SPEED * deltaTime; cameraUpdated = true;
99 if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS) positionDelta -= right * 2.0f * CAMERA_SPEED * deltaTime; cameraUpdated = true;
100 if (glfwGetKey(window, GLFW_KEY_E) == GLFW_PRESS) positionDelta += up * 2.0f * CAMERA_SPEED * deltaTime; cameraUpdated = true;
101 if (glfwGetKey(window, GLFW_KEY_Q) == GLFW_PRESS) positionDelta -= up * 2.0f * CAMERA_SPEED * deltaTime; cameraUpdated = true;
102
103 camera.position += positionDelta;
104 hri::Float3 target = camera.position + forward;
105
106 if (glfwGetKey(window, GLFW_KEY_UP) == GLFW_PRESS) target += up * CAMERA_SPEED * deltaTime; cameraUpdated = true;
107 if (glfwGetKey(window, GLFW_KEY_DOWN) == GLFW_PRESS) target -= up * CAMERA_SPEED * deltaTime; cameraUpdated = true;
108 if (glfwGetKey(window, GLFW_KEY_RIGHT) == GLFW_PRESS) target += right * CAMERA_SPEED * deltaTime; cameraUpdated = true;
109 if (glfwGetKey(window, GLFW_KEY_LEFT) == GLFW_PRESS) target -= right * CAMERA_SPEED * deltaTime; cameraUpdated = true;
110
111 if (!cameraUpdated)
112 return;
113
114 camera.forward = hri::normalize(target - camera.position);
115 camera.right = hri::normalize(hri::cross(HRI_WORLD_UP, camera.forward));
116 camera.up = hri::normalize(hri::cross(camera.forward, camera.right));
117}
