76VkRenderPass RenderPassBuilder::build()
77{
78 assert(m_pCtx != nullptr);
79
80 std::vector<VkSubpassDescription> subpasses = {}; subpasses.reserve(m_subpasses.size());
81 for (auto const& pass : m_subpasses)
82 {
83 VkSubpassDescription description = VkSubpassDescription{};
84 description.flags = 0;
85 description.colorAttachmentCount = static_cast<uint32_t>(pass.colorAttachments.size());
86 description.pColorAttachments = pass.colorAttachments.data();
87 description.pDepthStencilAttachment = pass.depthStencilAttachment.has_value() ? &pass.depthStencilAttachment.value() : nullptr;
88
89 subpasses.push_back(description);
90 }
91
92 VkRenderPassCreateInfo createInfo = VkRenderPassCreateInfo{ VK_STRUCTURE_TYPE_RENDER_PASS_CREATE_INFO };
93 createInfo.flags = 0;
94 createInfo.attachmentCount = static_cast<uint32_t>(m_attachments.size());
95 createInfo.pAttachments = m_attachments.data();
96 createInfo.subpassCount = static_cast<uint32_t>(subpasses.size());
97 createInfo.pSubpasses = subpasses.data();
98 createInfo.dependencyCount = 0;
99 createInfo.pDependencies = nullptr;
100
101 VkRenderPass newPass = VK_NULL_HANDLE;
102 HRI_VK_CHECK(vkCreateRenderPass(m_pCtx->device, &createInfo, nullptr, &newPass));
103
104 return newPass;
105}
