143Shader Shader::loadFile(RenderContext* ctx, const std::string& path, VkShaderStageFlagBits stage)
144{
145 // Open code file
146 FILE* file = fopen(path.c_str(), "rb");
147 if (file == nullptr)
148 {
149 fprintf(stderr, "Failed to open Shader File [%s]\n", path.c_str());
150 abort();
151 }
152
153 // Check file size
154 fseek(file, 0, SEEK_END);
155 long codeSize = ftell(file);
156 assert(codeSize > 0);
157
158 if (codeSize <= 0)
159 {
160 fclose(file);
161 fprintf(stderr, "Failed to read Shader File [%s]\n", path.c_str());
162 abort();
163 }
164
165 // Read shader binary blob
166 fseek(file, 0, SEEK_SET);
167 char* pCode = new char[codeSize + 1] {};
168 fread(pCode, sizeof(char), codeSize, file);
169
170 // Initialize shader
171 Shader shader = Shader::init(ctx, reinterpret_cast<uint32_t*>(pCode), static_cast<size_t>(codeSize), stage);
172
173 // Free resources
174 fclose(file);
175 delete[] pCode;
176
177 return shader;
178}
