195Shader Shader::loadFile(RenderContext& ctx, const std::string& path, VkShaderStageFlagBits stage)
196{
197 // Open code file
198 FILE* file = fopen(path.c_str(), "rb");
199 if (file == nullptr)
200 {
201 fprintf(stderr, "Failed to open Shader File [%s]\n", path.c_str());
202 abort();
203 }
204
205 // Check file size
206 fseek(file, 0, SEEK_END);
207 long codeSize = ftell(file);
208 assert(codeSize > 0);
209
210 if (codeSize <= 0)
211 {
212 fclose(file);
213 fprintf(stderr, "Failed to read Shader File [%s]\n", path.c_str());
214 abort();
215 }
216
217 // Read shader binary blob
218 fseek(file, 0, SEEK_SET);
219 char* pCode = new char[codeSize + 1] {};
220 size_t readBytes = fread(pCode, sizeof(char), codeSize, file);
221 assert(readBytes == codeSize);
222
223 // Initialize shader
224 Shader shader = Shader(ctx, reinterpret_cast<uint32_t*>(pCode), static_cast<size_t>(readBytes), stage);
225
226 // Free resources
227 fclose(file);
228 delete[] pCode;
229
230 return shader;
231}
