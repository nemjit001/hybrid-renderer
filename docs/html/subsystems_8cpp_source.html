166HybridRayTracingSubsystem::HybridRayTracingSubsystem(
167 raytracing::RayTracingContext& ctx,
168 hri::ShaderDatabase& shaderDB,
169 VkDescriptorSetLayout sceneDataSetLayout,
170 VkDescriptorSetLayout rtSetLayout
171)
172 :
173 IRayTracingSubSystem(ctx)
174{
175 m_layout = hri::PipelineLayoutBuilder(ctx.renderContext)
176 .addDescriptorSetLayout(sceneDataSetLayout)
177 .addDescriptorSetLayout(rtSetLayout)
178 .build();
179
180 const hri::Shader* pRayGen = shaderDB.getShader("HybridRayGen");
181 const hri::Shader* pRayMiss = shaderDB.getShader("HybridMiss");
182 const hri::Shader* pRayClosestHit = shaderDB.getShader("HybridClosestHit");
183
184 raytracing::RayTracingPipelineBuilder rtPipelineBuilder = raytracing::RayTracingPipelineBuilder(ctx)
185 .addShaderStage(pRayGen->stage, pRayGen->module)
186 .addShaderStage(pRayMiss->stage, pRayMiss->module)
187 .addShaderStage(pRayClosestHit->stage, pRayClosestHit->module)
188 .addRayTracingShaderGroup(VK_RAY_TRACING_SHADER_GROUP_TYPE_GENERAL_KHR, 0) // Ray Gen
189 .addRayTracingShaderGroup(VK_RAY_TRACING_SHADER_GROUP_TYPE_GENERAL_KHR, 1) // Ray Miss
190 .addRayTracingShaderGroup(VK_RAY_TRACING_SHADER_GROUP_TYPE_TRIANGLES_HIT_GROUP_KHR, VK_SHADER_UNUSED_KHR, 2) // Ray Hit
191 .setMaxRecursionDepth()
192 .setLayout(m_layout);
193
194 VkPipeline raytracingPipeline = rtPipelineBuilder.build(shaderDB.pipelineCache());
195 m_pPSO = shaderDB.registerPipeline(
196 "HybridRenderingRTPipeline",
197 VK_PIPELINE_BIND_POINT_RAY_TRACING_KHR,
198 raytracingPipeline);
199
200 // Init the SBT
201 m_SBT = std::unique_ptr<raytracing::ShaderBindingTable>(new raytracing::ShaderBindingTable(
202 m_rtCtx,
203 raytracingPipeline,
204 rtPipelineBuilder
205 ));
