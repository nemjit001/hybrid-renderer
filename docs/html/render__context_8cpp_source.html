160 .set_required_min_image_count(presentSetup.imageCount)
161 .set_image_usage_flags(oldSwapchain.image_usage_flags)

163 .build().value();

168SwapchainPresentSetup RenderContext::getSwapPresentSetup(VSyncMode vsyncMode)
169{
170 // Select image count & present mode based on VSync mode
171 uint32_t swapImageCount = 0;
172 VkPresentModeKHR presentMode = VK_PRESENT_MODE_FIFO_KHR;
173 switch (vsyncMode)
174 {
175 case VSyncMode::Disabled:
176 swapImageCount = HRI_DEFAULT_SWAP_IMAGE_COUNT;
177 presentMode = VK_PRESENT_MODE_MAILBOX_KHR;
178 break;
179 case VSyncMode::DoubleBuffering:
180 swapImageCount = 2;
181 presentMode = VK_PRESENT_MODE_FIFO_KHR;
182 break;
183 case VSyncMode::TripleBuffering:
184 swapImageCount = 3;
185 presentMode = VK_PRESENT_MODE_FIFO_KHR;
186 break;
187 }
188
189 return SwapchainPresentSetup{
190 swapImageCount,
191 presentMode,
192 };
193}
